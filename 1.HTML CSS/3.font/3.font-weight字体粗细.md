<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<title>Title</title>
<style>
.bold{
font-weight: bold;
/*字体加粗*/
}
.bo{
font-weight: 700;
/*等价于bold，都是加粗效果*/
/*实际开发中，更提倡用数字表示加粗或变细*/
}
.bolder{
font-weight: bolder;
/*更粗*/
}
h2{
font-weight: 400;
/*400等价于normal*/
}
</style>
</head>
<body>
<h2>标题</h2>
<p class="bold">Hello World</p>
<p class="bo">Hello World</p>
<p class="bolder">Hello World</p>
</body>
</html>
